var jsonData = pm.response.json()

pm.test("Response 'hasResult' is true", function () {
    pm.expect(jsonData.hasResult).to.be.true;
});

pm.test("Response 'result' is null", function () {
    pm.expect(jsonData.result.message).to.eql("Account status updated successfully");
});

pm.test("Response 'resultType' to eql 1'", function () {
    pm.expect(jsonData.resultType).to.equal(1);
});

pm.test("Response 'message' is a string and equals 'Account status updated successfully'", function () {
    pm.expect(pm.response.json().message).to.be.a('string').and.to.equal('Success');
});

pm.test("Response 'validationMessages' is null", function () {
    pm.expect(pm.response.json().validationMessages).to.be.null;
});

pm.test("Response 'successful' is true", function () {
    pm.expect(pm.response.json().successful).to.be.true;
});
