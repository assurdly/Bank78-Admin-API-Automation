pm.test("Status code is 404", function () {
    pm.response.to.have.status(404);
});

pm.test("Status code response is Not Found", function () {
    pm.response.to.have.status('Not Found');
});

var jsonData = pm.response.json();

pm.test("Response body", function () {
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData.hasResult).to.be.a('boolean');
    pm.expect(jsonData.result).to.be.null;
    pm.expect(jsonData.resultType).to.be.a('string');
    pm.expect(jsonData.successful).to.be.a('boolean');
});

pm.test("Response has a 'hasResult' property with value 'false'", function () {
    pm.expect(jsonData.hasResult).to.eql(false);
});

pm.test("Response has a 'resultType' property with value 'Error'", function () {
    pm.expect(jsonData).to.have.property("resultType", "Error");
});

pm.test("Response has a 'message' property with value 'Cannot find Business with specified RC/BN Number'", function () {
    pm.expect(jsonData).to.have.property("message", "Cannot find Business with specified RC/BN Number");
});

pm.test("Response has 'validationMessages'", function () {
    pm.expect(jsonData.validationMessages[0]).to.eq(null);
});

pm.test("Response has a 'successful' property with value 'false'", function () {
    pm.expect(jsonData).to.have.property(false);
});
